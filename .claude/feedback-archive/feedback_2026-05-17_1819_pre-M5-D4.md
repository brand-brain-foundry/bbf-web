/**
 * BBF Design System — HeroSection (section compound)
 *
 * Subordinado a: B-BBF-WEB-M5-D4-HEROSECTION
 * Decisiones: D-77 (surface), D-79 (compound), D-82 (AI-readable),
 *             D-83 (Atomic Design), D-88 (sections/), D-89 (compound API)
 *
 * Compound API canon BBF (D-79 aplica):
 *
 *   <HeroSection surface="dark">
 *     <HeroVideo>...</HeroVideo>            // children directos absolutamente posicionados
 *     <LocaleSwitcher />                     // children directos
 *     <HeroSection.Content>                  // contenido central
 *       <BBFLogo .../>
 *       <Heading level={1}>...</Heading>
 *       <Text>...</Text>
 *       <Button>...</Button>
 *     </HeroSection.Content>
 *   </HeroSection>
 *
 * Server Component (sin state interactivo).
 * Define surface context via data-surface attribute (D-77).
 */

import type { ReactNode } from 'react';
import { cn } from '@/lib/utils';
import {
  heroSectionVariants,
  heroSectionContentVariants,
  type HeroSectionVariants,
  type HeroSectionContentVariants,
} from './HeroSection.variants';

/* ============================================================
   ROOT
   ============================================================ */
export interface HeroSectionProps extends HeroSectionVariants {
  className?: string;
  children: ReactNode;
}

/**
 * BBF HeroSection — Section compound canon
 *
 * @description
 * Section principal del hero canon BBF. Compose atoms + molecules.
 * Define surface context (data-surface) para descendants surface-aware.
 *
 * @example Hero canon BBF
 * ```tsx
 * <HeroSection surface="dark">
 *   <HeroVideo autoplay muted loop>
 *     <HeroVideo.Source src="..." type="webm-vp9" />
 *     <HeroVideo.Source src="..." type="mp4-h264" />
 *   </HeroVideo>
 *   <LocaleSwitcher />
 *   <HeroSection.Content>
 *     <BBFLogo variant="stamp" size="hero" animated />
 *     <Heading level={1} variant="display-xl">
 *       we build<br />brand brains.
 *     </Heading>
 *     <Text variant="overline">Coming Soon</Text>
 *     <Button intent="cta" surface="dark" asChild>
 *       <a href="mailto:contacto@brandbrainfoundry.com">contact us</a>
 *     </Button>
 *   </HeroSection.Content>
 * </HeroSection>
 * ```
 */
function HeroSectionRoot({
  surface,
  height,
  className,
  children,
}: HeroSectionProps) {
  return (
    <section
      data-component="bbf-hero-section"
      data-surface={surface}
      className={cn(heroSectionVariants({ surface, height }), className)}
    >
      {children}
    </section>
  );
}

/* ============================================================
   CONTENT (sub-component)
   ============================================================ */
export interface HeroSectionContentProps extends HeroSectionContentVariants {
  className?: string;
  children: ReactNode;
}

/**
 * HeroSection.Content — Contenido central canon
 *
 * @description
 * Container central del hero. Spacing canon BBF (8pt grid via tokens hero.css).
 * z-20 sobre video bg.
 *
 * @example
 * ```tsx
 * <HeroSection.Content align="center">
 *   <BBFLogo ... />
 *   <Heading ...>...</Heading>
 * </HeroSection.Content>
 * ```
 */
function HeroSectionContent({
  align,
  className,
  children,
}: HeroSectionContentProps) {
  return (
    <div
      data-component="bbf-hero-section-content"
      className={cn(heroSectionContentVariants({ align }), className)}
    >
      {children}
    </div>
  );
}

/* ============================================================
   COMPOUND EXPORT (Object.assign canon Radix)
   ============================================================ */
export const HeroSection = Object.assign(HeroSectionRoot, {
  Content: HeroSectionContent,
});
```
